  def create_directories(m)
    BASEDIRS.each { |path| m.directory path }
    m.directory File.join('lib','sites')
    m.directory File.join('spec','isolation')
    m.directory File.join('spec','integration')
    m.directory File.join('spec','story')
    m.directory 'script'
  end
