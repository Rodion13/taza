    def create_entity_if_value_is_hash(key)
      if (@fixture.nil? && @hash[key].is_a?(Hash))
          Entity.new(@hash[key], nil)
      elsif (!@fixture.nil? && @hash[key].is_a?(Hash))
          Entity.new(@hash[key], @fixture)
      else
          @hash[key]
      end
    end
